class Solution { public int[] sumAndMultiply(String s, int[][] queries) { //first find substring // then use the function and substring sum, concat int[] result = new int[queries.length]; for (int i = 0; i < queries.length; i++) { int a = queries[i][0]; int b = queries[i][1]; String str = s.substring(a, b+1); result[i] = manipulation(str); } return result; } static final int MOD = 1_000_000_007; public int manipulation(String str) { long num = 0; long sum = 0; for (int i = 0; i < str.length(); i++) { char ch = str.charAt(i); if (ch != '0') { int digit = ch - '0'; sum += digit; num = (num * 10 + digit) % MOD; } } return (int)((num * (sum % MOD)) % MOD); } } -->
